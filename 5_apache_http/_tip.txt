INSTALL APACHE: from system yum: http://httpd.apache.org/docs/current/install.html

	sudo yum install httpd
	sudo systemctl enable httpd
	sudo systemctl start httpd

	SUSE makes it hard: - run this on apache to allow initiation of connections from apache
	/usr/sbin/setsebool -P httpd_can_network_connect 1

CREATE HTTP CONFIG for SERVICES

	touch /etc/httpd/conf.d/atlassian-vhost.conf
	vi /etc/httpd/conf.d/atlassian-vhost.conf


        <VirtualHost *:80>
            ServerName centos-linux-t
            Redirect 301 "/" "http://centos-linux-t/jira"

            ProxyRequests Off
            ProxyVia Off

            <Proxy *>
                 Require all granted
            </Proxy>

            ProxyPass /confluence http://172.17.8.182:8090/confluence
            ProxyPassReverse /confluence http://172.17.8.182:8090/confluence

            ProxyPass /jira http://172.17.8.181:8080/jira
            ProxyPassReverse /jira http://172.17.8.181:8080/jira

            ProxyPass /crowd http://172.17.8.187:8095/crowd
            ProxyPassReverse /crowd http://172.17.8.187:8095/crowd
        </VirtualHost>

    systemctl restart httpd
