CREATE SSL CONFIG and SELF-SIGNED CERTIFICATES:

CREATE CERTIFICATES:  https://wiki.centos.org/HowTos/Https

	yum install mod_ssl openssl
	openssl genrsa -out ca.key 2048
	openssl req -new -key ca.key -out ca.csr
	openssl x509 -req -days 365 -in ca.csr -signkey ca.key -out ca.crt
	cp ca.crt /etc/pki/tls/certs
	cp ca.key /etc/pki/tls/private/ca.key
	cp ca.csr /etc/pki/tls/private/ca.csr

EDIT SSL CONF FOR APACHE:

	vi +/SSLCertificateFile /etc/httpd/conf.d/ssl.conf
	SSLCertificateFile /etc/pki/tls/certs/ca.crt
	SSLCertificateKeyFile /etc/pki/tls/private/ca.key

	systemctl restart httpd

CREATE VIRTUAL HOST SSL CONFIG FOR ATLASSIAN TOOLS:

	 cp /etc/httpd/conf.d/atlassian-vhost.conf /etc/httpd/conf.d/atlassian-vhost-ssl.conf
     vi /etc/httpd/conf.d/atlassian-vhost-ssl.conf

	<VirtualHost *:443>
        SSLEngine on
        SSLCertificateFile /etc/pki/tls/certs/ca.crt
        SSLCertificateKeyFile /etc/pki/tls/private/ca.key

        ServerName centos-linux-t

        Redirect "/" "/jira"

        ProxyRequests Off
        ProxyVia Off

        <Proxy *>
                 Require all granted
        </Proxy>

        ProxyPass /confluence http://172.17.8.182:8090/confluence
        ProxyPassReverse /confluence http://172.17.8.182:8090/confluence

        ProxyPass /jira http://172.17.8.181:8080/jira
        ProxyPassReverse /jira http://172.17.8.181:8080/jira

        ProxyPass /crowd http://172.17.8.187:8095/crowd
        ProxyPassReverse /crowd http://172.17.8.187:8095/crowd
	</VirtualHost>

	systemctl restart httpd
